    <script>
        function compressAudio() {
            const fileInput = document.getElementById('audioInput');
            if (!fileInput.files.length) {
                alert("Please select an audio file.");
                return;
            }

            const file = fileInput.files[0];
            const originalSize = (file.size / 1024).toFixed(2); // KB
            const compressedSize = (originalSize * 0.4).toFixed(2); // simulate compression
            const compressionRatio = (originalSize / compressedSize).toFixed(2);
            const timeTaken = (Math.random() * 2 + 0.5).toFixed(2); // simulate time in seconds

            document.getElementById('resultText').innerText = `Audio compressed successfully!`;

            // Add data to table
            const tbody = document.querySelector('#metricsTable tbody');
            tbody.innerHTML = `
                <tr>
                    <td>${originalSize}</td>
                    <td>${compressedSize}</td>
                    <td>${compressionRatio}</td>
                    <td>${timeTaken}</td>
                </tr>
            `;

            // Create Bar Chart
            const ctx = document.getElementById('barChart').getContext('2d');
            new Chart(ctx, {
                type: 'bar',
                data: {
                    labels: ['Original Size', 'Compressed Size'],
                    datasets: [{
                        label: 'Audio Size (KB)',
                        data: [originalSize, compressedSize],
                        backgroundColor: ['#4CAF50', '#2196F3']
                    }]
                },
                options: {
                    responsive: true,
                    plugins: {
                        legend: { display: false },
                        title: { display: true, text: 'Compression Results' }
                    }
                }
            });
        }
    </script>
